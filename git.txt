1- git init # inialize the folder
2- git status # check the status of the folder
3- git add file1 file2 # add selected files from workspace to the stage
4- git add . # add all files in the current folder
5- git add --all # all files 
6- git commit -m "short brief Comment" # add the files to the master
7- git log # to see what is commit or not 
8- Upload from Local To Remote in GitHub and pull from remote to local
	- create the repo in github with the same name without createing readme files
	- git remote add origin https://github.com/elsayedseleim/ITI_Test_Git.git // link the local with master
	- git remote # to check the lists or remotes (only one origin) 
	- git remote -v # to see the origin point to what?
	- git push origin master #push the changes to remote will require the user and password for github
	- git pull origin master #pull the changes from remote to the local
